## ðŸ’» PROGRAM:
```python
class Program:
    def __init__(self, a, b, c):
        self.a = a   
        self.b = b   
        self.c = c   

    def display(self):
        
        even_numbers = [i for i in range(self.a, self.c + 1, self.b) if i % 2 == 0]
        print("Even numbers between", self.a, "and", self.c, "are:")
        print(even_numbers)

obj = Program(200, 1, 300)
obj.display()
